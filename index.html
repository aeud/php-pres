Recursive functions

<?php
function factorRec($n)
{
    if ($n == 1) {
        return 1;
    } else {
        return $n * factorRec($n - 1);
    }
    // return $n > 1 ? $n * factorRec($n - 1) : $n;
}
echo factorRec(13);
// 6227020800
?>
